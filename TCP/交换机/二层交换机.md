## 瑞斯康达交换机配置
::: details 点击展开
```js
Raisecom(config-ip)#show running-config

#由此开始
raisecom
raisecom
enable
config
create vlan 100,800 active
interface port 1
switchport access vlan 100
switchport trunk allowed vlan 100
y
!
interface port 2
switchport access vlan 100
switchport trunk allowed vlan 100
y
!
interface port 3
switchport access vlan 100
switchport trunk allowed vlan 100
y
!
interface port 4
switchport access vlan 100
switchport trunk allowed vlan 100
y
!
interface port 7
switchport access vlan 800
switchport mode trunk
switchport trunk allowed vlan 800
y
!
interface port 8
switchport access vlan 800
switchport mode trunk
switchport trunk allowed vlan 800
y
!
interface port 9
switchport access vlan 800
switchport mode trunk
switchport trunk allowed vlan 800
y
!
exit
interface ip 0
ip vlan 800
#在此添加静态路由（交换机地址）
ip address 192.169.26.11 255.255.255.0 
exit
exit
write
!
```
:::
## 思极交换机配置

::: details 点击展开
```js
#由此开始
！
admin
admin
information
config -n QZ
exit
vlan
enable 1
qvlansetting
add 100 1,2,7,8,9 u,u,t,t,t
y
add 800 0,6,7,8,9 u,u,t,t,t
y
config pvid 1-2 100
config pvid 0,6 800
config type 1-6 access
config type 7-9 trunk
exit
manage
#在此添加静态路由（交换机地址）
ip 192.169.11.66 255.255.255.0
#在此添加静态路由网关（上联上层交换机地址）
gateway 192.169.11.1
index 0
limit 1
username admin
password admin admin
exit
snmp
snmp-server trap enable
snmp-server host 192.168.8.6 public trapv2c
exit
ring
enable 4
stp 1
exit
lldp
enable 1
exit
port
enable 2,3,4,5, disable
description 7 QZHJJ-NKB-CE-SW01-GE1
description 8 QZHJJ-LGX#5HWG-CE-SW03-GE7
exit
save
！
```
:::
## 特发交换机配置

::: details 点击展开

```js
#由此开始
enable
config


vlan 100,800
!
lldp run
!
spanning-tree mode mstp
spanning-tree mstp max-hops 40
!
interface GigaEthernet0/1
switchport pvid 100
!
interface GigaEthernet0/2
 shutdown
!
interface GigaEthernet0/3
 shutdown
!
interface GigaEthernet0/4
 shutdown
!
interface GigaEthernet0/5
 shutdown
!
interface GigaEthernet0/6
 shutdown
!         
interface GigaEthernet0/7
 shutdown
!
interface GigaEthernet0/8
switchport pvid 800
!
interface GigaEthernet0/9
description To-QZHJJ-MLB-CE-SW01-IAC1724-12-DCB
 switchport mode trunk
 switchport trunk vlan-allowed 1,100,800
lldp management-ip 192.169.103.130
!
interface GigaEthernet0/10
 description To-QZHJJ-TMX#5HWG-CE-SW01-IAC1412-4G-B
 switchport mode trunk
 switchport trunk vlan-allowed 1,100,800
lldp management-ip 192.169.103.130
!
interface GigaEthernet0/11
 shutdown
!
interface GigaEthernet0/12
 shutdown
!
hostname -QZHJJ-TMX#4HWG-CE-SW01-IAC1412-4G-B
!
interface VLAN1
  no ip address
!
interface VLAN800
 ip address 192.169.103.130 255.255.255.192
!
ip default-gateway 192.169.103.129
!
username qzdl password qzdl0595
!
snmp-server community 0 public RO 
snmp-server host 192.168.8.6 version v2c public authentication configure snmp
!
no username admin
!
wr
!
```
